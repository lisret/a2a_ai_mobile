### 使用方式

```typescript
// 统一从模块入口导入
import { 
  capabilityService, 
  accessibilityService, 
  adbService,
  appMappingService,
  initializeCapabilities 
} from '@/abilityModule';

// 初始化
await initializeCapabilities();
await capabilityService.initialize();

// 使用
await capabilityService.captureScreen();
await accessibilityService.performClick(100, 200);
```
